from django.db import models
from django.contrib.auth.models import User

class Upload(models.Model):
    FILE_TYPE_CHOICES = [
        ('txt', 'Text File'),
        ('audio', 'Audio File'),
    ]

    TRANSLATION_STATUS = [
        ('pending', 'Pending'),
        ('success', 'Success'),
        ('failed', 'Failed'),
    ]

    user = models.ForeignKey(User, on_delete=models.CASCADE)
    original_file = models.FileField(upload_to='uploads/')
    file_type = models.CharField(max_length=10, choices=FILE_TYPE_CHOICES)
    
    original_text = models.TextField()  # Extracted text from file (txt or audio transcript)
    translated_text = models.TextField()
    target_language = models.CharField(max_length=50)

    pdf_file = models.FileField(upload_to='pdfs/', null=True, blank=True)

    status = models.CharField(max_length=10, choices=TRANSLATION_STATUS, default='pending')

    created_at = models.DateTimeField(auto_now_add=True)
    updated_at = models.DateTimeField(auto_now=True)

    def __str__(self):
        return f"{self.user.username} - {self.original_file.name} -> {self.target_language}"
